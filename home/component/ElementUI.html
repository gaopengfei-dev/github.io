# ElementUI

# 源码修改

首先访问github ElementUI 源码地址 (opens new window) 克隆一份 elementUI 代码

提示

若无法访问 github 可参考本平台其他类github 加速

目录结构

安装依赖

npm install

启动项目

npm run dev

在 packages 中找到对应的组件，根据需求修改源码

编译，执行

npm run dist

生成一个 lib 文件夹，将此文件夹替换项目中的 node_modules 中的 element-ui 文件夹中的 lib 文件即可

重新运行项目即可看到修改后的效果。

注

若项目重新安装依赖需重新替换 lib 文件夹

# 图片上传

使用方法

<img-upload :limit="1" v-model="pictureList" :filelist="pictureList">
  <span class="form-tip" slot="tip"
    >{{ formData.pictureType === 0 ?
    '首页轮播图建议尺寸1920*940，其他页面轮播图建议尺寸1920*590' :
    '首页轮播图建议尺寸375*340，其他页轮播图建议尺寸 375*340' }}</span
  >
</img-upload>

说明

上传使用$emit('input')方法，直接将数据插入到绑定的数组中；

# 参数配置

1、filelist

  • filelist 参数多用于修改页面图片的回显。
名称 类型 必填 默认值
filelist Array false null

2、isImgTitle

  • isImgTitle 控制标题的显示隐藏
名称 类型 必填 默认值
isImgTitle Boolean false false

3、imgTitleText

  • imgTitleText 标题的内容
名称 类型 必填 默认值
imgTitleText String false null

4、limit

  • limit 控制标题的显示隐藏
名称 类型 必填 默认值
limit Number false null

代码

<template>
  <div class="">
    <span v-if="isImgTitle">{{ imgTitleText }}：</span>
    <el-upload
      :class="limitJudge ? 'no-upload-btn' : ''"
      list-type="picture-card"
      action="#"
      :before-upload="handleBefore"
      :http-request="httpRequest"
      :file-list="fileList"
      accept="image/*"
      :limit="limit"
    >
      <i class="el-icon-plus"></i>
      <div class="w_100_h_100" slot="file" slot-scope="{ file }">
        <img
          class="el-upload-list__item-thumbnail w_100_h_100"
          :src="file.fileUrl"
        />
        <span class="el-upload-list__item-actions">
          <span
            class="el-upload-list__item-preview"
            @click="handlePictureCardPreview(file)"
          >
            <i class="el-icon-zoom-in"></i>
          </span>
          <span class="el-upload-list__item-delete" @click="handleRemove(file)">
            <i class="el-icon-delete"></i>
          </span>
        </span>
      </div>
    </el-upload>
    <slot name="tip"></slot>
    <el-dialog :visible.sync="dialogVisible" :modal-append-to-body="false">
      <img width="100%" :src="dialogImageUrl" />
    </el-dialog>
  </div>
</template>

<script>
import { uploadFile } from '@/http/public';
// import 'viewerjs/dist/viewer.css';
// import VueViewer from 'v-viewer';
// import Vue from 'vue';
// Vue.use(VueViewer, {
//   defaultOptions: {
//     zIndex: 9999
//   }
// });
export default {
  props: [
    'filelist',
    'isImgTitle',
    'imgTitleText',
    'limit',
  ],
  data() {
    return {
      fileList: [],
      fileType: 1,
      listArr: [],
      dialogImageUrl: '',
      dialogVisible: false
    };
  },
  watch: {
    filelist(val) {
      if (val && val.length > 0) {
        val.forEach(e => {
          e.name = e.fileRealName;
        });
        this.fileList = val;
        this.listArr = val;
      }
    }
  },
  computed: {
    limitJudge() {
      return this.fileList.length === this.limit;
    }
  },
  methods: {
    // 文件上传处理
    httpRequest(options) {
      let that = this;
      let fileFormData = new FormData();
      fileFormData.append('file', options.file, options.file.name); //filename是键，file是值，就是要传的文件，test.zip是要传的文件名
      fileFormData.append('fileType', that.fileType);
      uploadFile(fileFormData).then(res => {
        if (res.data.code === 200) {
          let resdata = res.data;
          if (resdata) {
            that.listArr.push({
              fileUrl: resdata.data[0].fileUrl,
              fileId: resdata.data[0].fileId,
              fileType: resdata.data[0].fileType,
              fileName: resdata.data[0].fileName,
              fileRealName: resdata.data[0].fileRealName,
              uid: options.file.uid
            });
            that.$emit('input', that.listArr);
          } else {
            this.$message.error(resdata.message);
          }
        }
      });
    },
    // 上传文件之前处理
    handleBefore(file) {
      const fileType = file.name
        .substring(file.name.lastIndexOf('.') + 1)
        .toLowerCase();
      const isLt2M = file.size / 1024 / 1024 < 5;
      const isJPG = fileType === 'jpg';
      const isJPEG = fileType === 'jpeg';
      const isPNG = fileType === 'png';
      if (!isJPG && !isJPEG && !isPNG) {
        this.$message.error('请上传.jpg .jpeg .png类型文件');
        return false;
      }
      if (!isLt2M) {
        this.$message.error('请上传不大于5M的图片');
        return false;
      }
    },
    // 删除文件
    handleRemove(file) {
      this.$confirm(`确定移除 ${file.name}？`, '', {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning',
        showClose: false
      }).then(() => {
        this.listArr.forEach((v, i) => {
          if (v.uid === file.uid) {
            this.listArr.splice(i, 1);
            this.fileList.splice(i, 1);
          }
        });
        this.$emit('input', this.listArr);
      });
    },
    // 图片查看
    handlePictureCardPreview(file) {
      this.$viewerApi({
        images: [file.fileUrl]
      });
    }
  }
};
</script>
