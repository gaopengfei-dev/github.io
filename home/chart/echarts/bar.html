# 仿 3D 柱状图

<template>
  <div id="chart" ref="chart"></div>
</template>

<script>
import * as echarts from "echarts";
export default {
  data() {
    return {};
  },
  mounted() {
    this.$nextTick(() => {
      this.initFn();
    });
  },
  methods: {
    initFn() {
      let _this = this;
      let myChart = echarts.init(_this.$refs.chart);
      myChart.setOption({
        xAxis: {
          data: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
        },
        yAxis: {},
        series: [
          {
            type: "bar",
            name: "销量",
            barWidth: 30,
            itemStyle: {
              normal: {
                color: {
                  type: "linear",
                  x: 0,
                  y: 1,
                  x2: 0,
                  y2: 0,
                  colorStops: [
                    {
                      offset: 0,
                      color: "rgba(61,126,235,0.4)", // 0% 处的颜色
                    },
                    {
                      offset: 1,
                      color: "rgba(61,126,235,1)", // 100% 处的颜色
                    },
                  ],
                  globalCoord: false, // 缺省为 false
                },
              },
            },
            data: [5, 20, 36, 10, 10, 20],
          },
          {
            type: "bar",
            barGap: 0,
            barWidth: 6,
            itemStyle: {
              normal: {
                color: {
                  type: "linear",
                  x: 0,
                  y: 1,
                  x2: 0,
                  y2: 0,
                  colorStops: [
                    {
                      offset: 0,
                      color: "rgba(61,126,235,0.2)", // 0% 处的颜色
                    },
                    {
                      offset: 1,
                      color: "rgba(61,126,235,1)", // 100% 处的颜色
                    },
                  ],
                  globalCoord: false, // 缺省为 false
                },
              },
            },
            data: [5, 20, 36, 10, 10, 20],
          },
        ],
      });
    },
  },
};
</script>

# 渐变仪表盘

<!--
 * @Description: 
 * @Author: 郜鹏飞
 * @Date: 2021-10-25 16:37:55
 * @LastEditTime: 2021-11-04 10:46:37
 * @LastEditTime: 2021-10-27 10:37:24
-->
<template>
  <div class="main"></div>
</template>

<script>
import * as echarts from 'echarts';
export default {
  props: {
    // 动态id
    chartId: {
      type: String
    },
    // 总数
    total: {
      type: Number,
      default: 300
    },
    // 当前数值
    currentVal: {
      type: Number,
      default: 120
    }
  },
  data() {
    return {
      // 定规渐变颜色
      color1: [
        { offset: 0.25, color: '#299DFF' },
        { offset: 0.5, color: '#20E3B2' },
        { offset: 0.75, color: '#0CEBEB' },
        { offset: 1, color: '#0CEBEB' }
      ],
      color2: [
        { offset: 0.5, color: '#FF8008' },
        { offset: 1, color: '#FFC837' }
      ],
      color3: [
        { offset: 0.5, color: '#2966F1' },
        { offset: 1, color: '#A939FF' }
      ],
      // 刻度颜色
      axisColor1: [
        [0.25, '#299DFF'],
        [0.5, '#20E3B2'],
        [0.75, '#0CEBEB'],
        [1, '#0CEBEB']
      ],
      axisColor2: [
        [0.25, '#FFC837'],
        [1, '#FF8008']
      ],
      axisColor3: [
        [0.5, '#A939FF'],
        [1, '#2966F1']
      ]
    };
  },
  mounted() {
    this.getCharts();
  },
  methods: {
    //  获取主题样式
    getColor(id) {
      if (id === 'mcf_gauge1') {
        return this.color1;
      } else if (id === 'mcf_gauge2') {
        return this.color2;
      } else {
        return this.color3;
      }
    },
    // 获取样式
    getAxisColors(id) {
      if (id === 'mcf_gauge1') {
        return this.axisColor1;
      } else if (id === 'mcf_gauge2') {
        return this.axisColor2;
      } else {
        return this.axisColor3;
      }
    },
    getCharts() {
      const self = this;
      var chartDom = document.getElementById(this.chartId);
      var myChart = echarts.init(chartDom);
      let option;
      option = {
        series: [
          {
            type: 'gauge',
            min: 0,
            max: 100,
            splitNumber: 5,
            center: ['50%', '50%'],
            itemStyle: {
              color: {
                type: 'linear',
                x: 0,
                y: 0,
                x2: 1, //从左到右 0-1
                y2: 0,
                colorStops: self.getColor(self.chartId)
              }
            },
            title: {
              show: true,
              offsetCenter: [0, '105%'],
              textStyle: {
                fontWeight: '400',
                fontSize: 16,
                color: 'auto'
              }
            },
            progress: {
              show: true,
              width: 16,
              roundCap: true
            },
            axisLine: {
              lineStyle: {
                color: self.getAxisColors(self.chartId),
                opacity: 0
              }
            },
            axisTick: {
              show: true,
              length: 10,
              distance: -35,
              splitNumber: 10,
              lineStyle: {
                color: 'auto'
              }
            },
            splitLine: {
              show: true,
              distance: -40,
              length: 15,
              lineStyle: {
                width: 2,
                color: 'auto'
              }
            },
            axisLabel: {
              distance: 50,
              color: 'auto',
              fontSize: 14
            },
            anchor: {
              show: true,
              showAbove: true,
              size: 15,
              itemStyle: {
                borderWidth: 0,
                color: '#21A1E7'
              }
            },

            detail: {
              show: true,
              valueAnimation: true,
              fontSize: '22px',
              color: self.getAxisColors(self.chartId)[0][1],
              offsetCenter: [0, '55%'],
              formatter: `${self.currentVal}\n {time|生产量（公斤）}`,
              rich: {
                time: {
                  fontSize: 12,
                  color: '#fff'
                }
              }
            },

            data: [
              {
                value: Math.floor((self.currentVal / self.total) * 100),
                name: '计划产量：' + self.total
              }
            ]
          },
          // 底层深色环
          {
            type: 'gauge',
            center: ['50%', '50%'],
            progress: {
              show: false
            },
            axisLine: {
              lineStyle: {
                width: 16,
                color: [[1, '#14306F']],
                opacity: 1
              },
              roundCap: true
            },
            axisTick: {
              show: false
            },
            splitLine: {
              show: false
            },
            axisLabel: {
              show: false
            },
            anchor: {
              show: false
            },
            title: {
              show: true
            }
          }
        ]
      };

      option && myChart.setOption(option);
      window.addEventListener('resize', () => {
        myChart.resize();
      });
    }
  }
};
</script>

  height: 330px;
  background-color: #071642;

  box-sizing: border-box;
