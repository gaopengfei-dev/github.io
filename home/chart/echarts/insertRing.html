<script>
import * as echarts from 'echarts';
export default {
  props: {
    chartId: {
      type: String
    }
  },
  data() {
    return {};
  },
  mounted() {
    this.getCharts();
  },
  methods: {
    getCharts() {
      let jihua = 324780,
        dingdan = 242110,
        shengchan = 67321;

      var chartDom = document.getElementById(this.chartId);
      var myChart = echarts.init(chartDom);
      let option = {
        tooltip: {
          show: true,

          trigger: 'item',
          formatter: '{a0} : {c0} ({d0}%)',

          borderColor: '#00479D',
          textStyle: {
            color: '#fff'
          },
          extraCssText: 'box-shadow: 0 0 3px #173C90;' //添加阴影
        }, // 设为不显示了，保存了基本样式
        color: ['rgb(255,159,127)', 'rgb(50,197,233)'],
        legend: {
          show: false
        },
        title: {
          text: '标题',
          left: 'center',
          bottom: '0%',
          textStyle: {
            color: '#fff',
            fontSize: 14
          }
        },
        series: [
          {
            name: '计划量',
            type: 'pie',
            radius: ['70%', '85%'],
            hoverAnimation: false,
            //环的位置
            label: {
              show: false,
              position: 'center'
            },
            labelLine: {
              normal: {
                show: false
              }
            },
            emphasis: {
              label: {
                show: false,
                fontSize: '20',
                fontWeight: 'bold'
              }
            },

            data: [
              {
                value: 324780, //需要显示的数据
                name: '计划量',
                itemStyle: {
                  normal: {
                    color: {
                      type: 'linear',
                      x: 1,
                      y: 1,
                      x2: 0,
                      y2: 0,
                      colorStops: [
                        { offset: '0', color: '#F2994A' },
                        { offset: '1', color: '#F2C94C' }
                      ]
                    }
                  }
                }
              }
            ]
          },
          {
            type: 'pie',
            hoverAnimation: false,
            radius: ['63%', '64%'],
            tooltip: {
              show: false
            },
            label: {
              show: false,
              position: 'center'
            },
            data: [
              {
                value: 100,
                itemStyle: {
                  normal: {
                    color: 'RGBA(25, 84, 161, 1)'

                }
              }
            ]
          }, // 分界线
          {
            name: '订单量',
            type: 'pie',
            hoverAnimation: false,
            radius: ['45%', '58%'],
            label: {
              show: false,
              position: 'center'
            },
            labelLine: {
              normal: {
                show: false
              }
            },
            emphasis: {
              label: {
                show: false,
                fontSize: '20',
                fontWeight: 'bold'
              }
            },

            data: [
              {
                name: '订单量',
                value: dingdan,
                itemStyle: {
                  normal: {
                    color: {
                      colorStops: [
                        { offset: '0', color: '#4776E6' },

              },

                value: jihua - dingdan,

      });
    }
  }
